% ── Frame 2: Problem — The Risk Nobody Hedges ────────────────────────────────
\begin{frame}{The Risk Nobody Hedges}

In a symmetric equilibrium, each of $\PosCount$ LPs earns $1/\PosCount$ of
fees \citep{macrapis2024permissionless}. In practice, sophisticated actors
concentrate fee revenue away from passive participants.
This is \textbf{adverse competition} --- a distinct, unhedged risk.

\bigskip

\begin{itemize}
  \item \textbf{Impermanent loss (IL)} --- price divergence between pool assets.
    Variance perpetuals perfectly hedge IL
    \citep{fukasawa2023weighted}; Panoptic deploys this on-chain.

  \item \textbf{Loss-versus-rebalancing (LVR)} --- adverse selection cost from
    informed traders. MEV-aware products address LVR.

  \item \textbf{Fee concentration risk} --- the \emph{competition structure}
    among LPs for fee revenue, captured by the FLAIR metric
    \citep{milionis2024flair}.
    \textcolor{red}{\textbf{Nothing hedges this today.}}
\end{itemize}

\end{frame}

% ── Frame 3: Research Question ────────────────────────────────────────────────
\begin{frame}{Research Question}

\textbf{Is there empirical demand for a hedging instrument against adverse
competition? If so, how much are LPs willing to pay?}

\bigskip

\begin{block}{Data}
ETH/USDC 30bps pool (Uniswap V3 mainnet) \\
41 days $\cdot$ 600 positions $\cdot$ 3,365 position-day observations $\cdot$ 597 exits
\end{block}

\bigskip

\textbf{Treatment variable --- concentration deviation:}
\begin{center}
\boxed{
  \DeltaPlus = \max\!\bigl(0,\; \AT - \ATnull\bigr)
}
\end{center}

\begin{itemize}
  \item $\AT = \bigl(\sum_k \thetapos_k \cdot x_k^2\bigr)^{1/2}$ \quad
    fee concentration index ($x_k$ = fee share, $\thetapos_k = 1/\text{blocklife}$)
  \item $\ATnull = \sqrt{\ThetaSum / \PosCount^2}$ \quad
    Ma--Crapis competitive null (equal shares $x_k = 1/\PosCount$)
  \item $\DeltaPlus > 0$: fees more concentrated than the symmetric equilibrium
\end{itemize}

\end{frame}

% ── Frame 4: Model — The Shelter Component ───────────────────────────────────
\begin{frame}{Model: The Shelter Component}

\begin{center}
$P(\text{exit}_{i,t}) = \sigma\!\bigl(
  \beta_0
  + \beta_1 \,\textcolor{red}{\boxed{\Delta_{t-\ell}}}
  + \beta_2 \,\Delta_{t-\ell}^2
  + \beta_3 \,\text{IL}_t
  + \beta_4 \log(\text{age}_{i,t})
\bigr)$
\end{center}

\bigskip

\textbf{We expect $\beta_1 < 0$} (shelter effect):

\begin{itemize}
  \item Mild fee concentration correlates with \emph{high trading volume}
    --- the pool is active, all LPs benefit from elevated fee revenue

  \item At low $\Delta$, more concentration means more volume,
    which \emph{lowers} exit probability

  \item This is \textbf{healthy competition}: sophisticated LPs attract flow
    that benefits everyone \citep{capponi2024jit}

  \item A linear-only model would conclude concentration is \emph{protective}
    --- missing the adverse regime entirely
\end{itemize}

\end{frame}

% ── Frame 5: Model — The Adverse Competition Component ──────────────────────
\begin{frame}{Model: The Adverse Competition Component}

\begin{center}
$P(\text{exit}_{i,t}) = \sigma\!\bigl(
  \beta_0
  + \beta_1 \,\Delta_{t-\ell}
  + \beta_2 \,\textcolor{green!50!black}{\boxed{\Delta_{t-\ell}^2}}
  + \beta_3 \,\text{IL}_t
  + \beta_4 \log(\text{age}_{i,t})
\bigr)$
\end{center}

\bigskip

\textbf{We expect $\beta_2 > 0$} (adverse competition):

\begin{itemize}
  \item Beyond a threshold, concentration no longer signals volume ---
    it signals \emph{fee extraction} by sophisticated actors

  \item The quadratic term $\Delta^2$ captures this \textbf{nonlinear regime shift}:
    the marginal effect of concentration \emph{flips sign} at the turning point

  \item Together: $\beta_1 < 0$ + $\beta_2 > 0$ $=$ \textbf{inverted-U} with
    turning point $\DeltaStar = -\beta_1 / (2\beta_2)$
\end{itemize}

\end{frame}

% ── Frame 7: Evidence — What the Data Says ───────────────────────────────────
\begin{frame}{What the Data Says}

\textit{Passive LPs respond to concentration shocks within one to three days.
When fee dispersion exceeds
${\sim}9\%$ above the competitive equilibrium
{\footnotesize ($\DeltaStar = -\beta_1/2\beta_2 = 23.18/(2 \times 129.20) = 0.090$)},
the shelter effect reverses: each additional percentage point of concentration
raises exit probability by approximately 2.3 percentage points
{\footnotesize ($(\beta_1 + 2\beta_2 \cdot 0.15)\,\bar{P}(1{-}\bar{P}) = 15.58 \times 0.146 = 2.27$)}.
At a fee dispersion level of 15\% above equilibrium, LPs in this pool are
willing to pay approximately \textbf{10\% above the average hourly pool fee
revenue} for protection
{\footnotesize ($2.3\text{pp} \times 48\text{h avg remaining} \approx 1.1\text{h of fees}$, or ${\sim}10\%$ of hourly revenue)}.}

\bigskip

This establishes both the \textbf{existence of demand} (the inverted-U with
a statistically significant adverse regime) and a \textbf{concrete price}
(the marginal effect translated to an insurance premium).

\end{frame}

% ── Frame: Solution — FCI Algorithm (Tracking) ──────────────────────────────
\begin{frame}{FCI Algorithm: Position Tracking}

\begin{block}{\textsc{AfterAddLiquidity}$(sender, key, params)$}
\ttfamily\small
1: $posKey \gets$ Facet.positionKey$(sender, params)$ \\
2: Facet.addPositionInRange$(posKey, snapshot)$ \\
3: $baseline_k \gets$ Facet.poolRangeFeeGrowthInside$(poolId, tick, range)$ \\
4: Facet.setFeeGrowthBaseline$(poolId, posKey, baseline_k)$ \\
5: $\PosCount \gets \PosCount + 1$
\end{block}

\begin{block}{\textsc{BeforeSwap / AfterSwap}$(key)$}
\ttfamily\small
1: $tick_{\text{before}} \gets$ Facet.currentTick$(poolId)$ \\
2: Facet.tstoreTick$(tick_{\text{before}})$ \hfill\textrm{\itshape\% transient storage} \\
3: $tick_{\text{after}} \gets$ Facet.currentTick$(poolId)$ \hfill\textrm{\itshape\% after swap} \\
4: Facet.incrementOverlappingRanges$(poolId, tick_{\min}, tick_{\max})$
\end{block}

\medskip

{\footnotesize All calls are \texttt{delegatecall} to the registered protocol facet.
Protocols implement \texttt{IFCIProtocolFacet} --- same algorithm, any AMM.}

\end{frame}

% ── Frame: Solution — FCI Algorithm (Accumulation) ──────────────────────────
\begin{frame}{FCI Algorithm: Index Accumulation}

\begin{block}{\textsc{BeforeRemoveLiquidity}$(sender, key, params)$}
\ttfamily\small
1: $(posLiq, feeLast_k) \gets$ Facet.latestFeeGrowth$(poolId, posKey)$ \\
2: $rangeFee \gets$ Facet.poolRangeFeeGrowthInside$(poolId, tick, range)$ \\
3: Facet.tstoreRemovalData$(feeLast_k, posLiq, rangeFee)$
\end{block}

\begin{block}{\textsc{AfterRemoveLiquidity}$(sender, key, params)$}
\ttfamily\small
1: \textbf{if} partial remove \textbf{then return} \hfill\textrm{\itshape\% skip accumulation} \\
2: $(swapLife, blockLife, totalLiq) \gets$ Facet.removePosition$(posKey)$ \\
3: \textbf{if} $swapLife > 0$ \textbf{then}: \\
\quad 4: $x_k \gets$ fromFeeGrowthDelta$(rangeFee, feeLast_k, baseline_k, posLiq, totalLiq)$ \\
\quad 5: $\thetapos_k \gets 1 / blockLife$ \\
\quad 6: $\AT^2 \mathrel{+}= \thetapos_k \cdot x_k^2$ \hfill\textrm{\itshape\% state accumulation} \\
7: $\PosCount \gets \PosCount - 1$
\end{block}

\medskip

{\footnotesize Source: \texttt{FeeConcentrationIndexV2.sol}.
Query: $\DeltaPlus = \max(0, \sqrt{\AT^2} - \ATnull)$ via \texttt{getDeltaPlus()}.}

\end{frame}

% ── Frame: Solution — Metrics Interface ──────────────────────────────────────
\begin{frame}[fragile]{Metrics Interface}

What an LP or indexer can query via \texttt{FeeConcentrationIndexV2}:

\medskip

{\small
\begin{tabular}{@{}ll@{}}
\toprule
\textbf{Function} & \textbf{Returns} \\
\midrule
\texttt{getIndex(key, flags)} & $\AT$, $\ThetaSum$, removed position count \\
\texttt{getDeltaPlus(key, flags)} & $\DeltaPlus = \max(0, \sqrt{\AT^2} - \ATnull)$ \\
\texttt{getAtNull(key, flags)} & $\ATnull = \sqrt{\ThetaSum / \PosCount^2}$ \\
\texttt{getThetaSum(key, flags)} & $\ThetaSum = \sum_k \thetapos_k$ \\
\texttt{getDeltaPlusEpoch(key, flags)} & $\DeltaPlus$ for current epoch window \\
\midrule
\texttt{getRegistryActiveRanges(key, flags)} & all tracked tick ranges \\
\texttt{getRegistryAllSnapshots(key, flags)} & range-level snapshots (liq, swaps, fees) \\
\texttt{getRegistryPositionBaseline(key, flags, posKey)} & fee growth baseline at entry \\
\bottomrule
\end{tabular}
}

\bigskip

{\footnotesize
\textbf{Note:} Epochs are essential --- the cumulative $\AT$ has infinite memory
and grows monotonically. Without epoch windowing, the index reflects the entire
pool history rather than recent concentration. \texttt{getDeltaPlusEpoch()}
provides the time-bounded metric.

\medskip

Interface: \texttt{interfaces/IFCIProtocolFacet.sol} \\
Reference implementation: \texttt{protocols/uniswap-v4/NativeUniswapV4Facet.sol}}

\end{frame}
